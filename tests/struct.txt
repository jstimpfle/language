struct bar {
    data int x;
    data int y;
    data int z;
}

struct foo {
    data int x;
    data bar y;
}

proc int main()
{
    data foo a;
    a.x = 3;
    a.y.x = 4;
    (&a.y.x)[1] = 5;
    a.y.z = 6;
    print64(a.x);
    print64(a.y.x);
    print64(a.y.y);
    print64(a.y.z);
    return 0;
}

export main;
